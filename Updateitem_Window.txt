    private void checkActionPerformed(java.awt.event.ActionEvent evt) {                                      
        
        String Item_id = item_id.getText();
   
        try{
            Connection Con = connector_mysql.getCon();
            Statement st = Con.createStatement();
            ResultSet rs = st.executeQuery("select * from stock where Item_id = '"+Item_id+"' ");
            if(rs.next()){
                category.setText(rs.getString(2));
                name.setText(rs.getString(3));
                category.setEditable(false);
                name.setEditable(false);
            }
            else{
                JOptionPane.showMessageDialog(null,"Incorrect Item_id");
                setVisible(false);
                new UpdateItemQuantity_Window().setVisible(true);
            }
        }
        catch(Exception e){
            JOptionPane.showMessageDialog(null,"Incorrect Item_id");
        }
    }

    private void updateActionPerformed(java.awt.event.ActionEvent evt) {                                       
        String Item_id = item_id.getText();
        int Quantity = Integer.parseInt(jTextField1.getText());
   
        try{
            Connection Con = connector_mysql.getCon();
            Statement st = Con.createStatement();
            st.executeUpdate("UPDATE stock SET Quantity = Quantity + '"+Quantity+"' WHERE Item_id = '"+Item_id+"'");
            st.executeQuery("Commit");
            JOptionPane.showMessageDialog(null, "Updated Successfully");
            setVisible(false);
            new UpdateItemQuantity_Window().setVisible(true);
            
        }
        catch(Exception e){
            JOptionPane.showMessageDialog(null, "Item_id doesn't exist");
        }
    }
