    private void addbutton1ActionPerformed(java.awt.event.ActionEvent evt) {                                           
        
        quantity = Integer.parseInt(quantity1.getSelectedItem().toString());
        nameofitem = vaccinescombobox.getSelectedItem().toString();
        try{
            Connection Con = connector_mysql.getCon();
            Statement st = Con.createStatement();
            ResultSet rs = st.executeQuery("select Category,Price from stock where Name = '"+nameofitem+"' ");
            if(rs.next()){
                category = rs.getString(1);
                amount = Integer.parseInt(rs.getString(2))*quantity;
                total_price+=amount;
                totalamounttextfield.setText(Integer.toString(total_price));
            }
        }
        catch(Exception e){
            System.out.println(e);
            JOptionPane.showMessageDialog(null,"Item_id already exists");
            
        }
        DefaultTableModel tb1 = (DefaultTableModel)orderdetailstable.getModel();
        String data[] = {category,nameofitem,Integer.toString(quantity),Integer.toString(amount)};
        tb1.addRow(data);
    }                                          

    private void placeorderbuttonActionPerformed(java.awt.event.ActionEvent evt) {                                                 
        
        JOptionPane.showMessageDialog(null,"Please pay Rs "+total_price);
        nameofperson = Nameofperson.getText();
        mobileno = Mobileno.getText();
        email = Email.getText();
        adress = Adress.getText();
        String date = java.time.LocalDate.now().toString();
        String time = java.time.LocalTime.now().toString().substring(0, 8);
        designation = "Hospital";
        try{
            Connection Con = connector_mysql.getCon();
            Statement st = Con.createStatement();
            st.executeUpdate("insert into history values('"+date+"','"+time+"','"+designation+"','"+nameofperson+"','"+mobileno+"','"+email+"','"+adress+"','"+total_price+"')");
            int i = orderdetailstable.getRowCount();
            for(int j=0;j<i;j++){
                String n = orderdetailstable.getModel().getValueAt(j, 1).toString();
                int q = Integer.parseInt(orderdetailstable.getModel().getValueAt(j, 2).toString());
                st.executeUpdate("UPDATE stock SET Quantity = Quantity - '"+q+"' WHERE Name = '"+n+"'");
            }

        }
        catch(Exception e){
            System.out.println(e);
        }
        setVisible(false);
        new Hospital_Window().setVisible(true);
        
        
        
        
       
    }

    void getItemnames(){
        
        try{
            Connection Con = connector_mysql.getCon();
            Statement st = Con.createStatement();
            ResultSet rs = st.executeQuery("Select Name from stock");
            while(rs.next()){
                String name = rs.getString("Name");
                vaccinescombobox.addItem(name);
            }
        }
        catch(Exception e){
            System.out.println(e);
