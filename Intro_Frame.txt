    private void proceedbuttonActionPerformed(java.awt.event.ActionEvent evt) {                                              
        desg = desg_combobox.getSelectedItem().toString();
        if(desg_combobox.getSelectedItem()=="Shop Owner(Administrator)"){
            setVisible(false);
            new ShopOwner_Window().setVisible(true);
        }
        else if(desg_combobox.getSelectedItem()=="Customer(Patient)"){
            setVisible(false);
            new Customer_Window().setVisible(true);
        }
        else if(desg_combobox.getSelectedItem()=="Hospital Manager"){
            setVisible(false);
            new Hospital_Window().setVisible(true);
        }
    }
