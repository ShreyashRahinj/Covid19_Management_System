    private void proceedbuttonActionPerformed(java.awt.event.ActionEvent evt) {                                              
        if(buttongroup.getSelection() == viewstockbutton.getModel()){
            setVisible(false);
            new ViewStock_Window().setVisible(true);
        }
        if(buttongroup.getSelection() == viewtransactionbutton.getModel()){
            setVisible(false);
            new ViewTransaction_Window().setVisible(true);
        }
        if(buttongroup.getSelection() == updatebutton.getModel()){
            setVisible(false);
            new UpdateItemQuantity_Window().setVisible(true);
        }
        if(buttongroup.getSelection() == addbutton.getModel()){
            setVisible(false);
            new AddItem_Window().setVisible(true);
        }
    }
